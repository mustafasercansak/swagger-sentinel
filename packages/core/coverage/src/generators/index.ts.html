export function generate(

	options: { baseUrl?: string; tag?: string; output?: string } = {},
): Array<{ name: string; content: string }> {
	const ops = getAllOperations(spec);
	const baseUrl = options.baseUrl || "http://localhost:3000";
	const tagFilter = options.tag || null;
 
	// Group operations by tag
	const byTag: Record<string, Array<(typeof ops)[0]>> = {};
	for (const op of ops) {
		const tags = op.operation.tags || ["untagged"];
		for (const tag of tags) {
			if (tagFilter && tag.toLowerCase() !== tagFilter.toLowerCase()) continue;
			if (!byTag[tag]) byTag[tag] = [];
			byTag[tag].push(op);
		}
	}
 
	const files: Array<{ name: string; content: string }> = [];
 
	// Generate helper file
	files.push({
		name: "api-helper.ts",
		content: generateHelper(baseUrl),
	});
 
	// Generate schema file
	files.push({
		name: "schemas.ts",
		content: generateSchemas(spec),
	});
 
	// Generate test file per tag
	for (const [tag, tagOps] of Object.entries(byTag)) {
		const fileName = `${toKebabCase(tag)}.test.ts`;
		const content = generateTestFile(tag, tagOps, spec);
		files.push({ name: fileName, content });
	}
 
	return files;
}

function toKebabCase(str: string) {
	return str

		.replace(/[^a-zA-Z0-9]+/g, "-")
